81 /** @brief The run method that runs the user interface thread.
82  * @details This method checks for characters typed by the user and responds to any
83  * typed characters.
84  * @return This method should never return; if it hypothetically did, it would
85  * return an OS message of zero.
86  */
87 
88 void task_user::run (void)
89 {
90  char ch_in; // A character sent to us by the user
91  uint16_t temp_num; // Temporary storage for a number
92  uint8_t oversampling; // How many times to oversample if > 0
93 
94  for (;;)
95  {
96  if (p_serial->check_for_char ())
97  {
98  ch_in = p_serial->getchar ();
99 
100  // Reading A/D channels 2 and 3 isn't allowed because it messes up the
101  // serial port by reconfiguring serial I/O lines into A/D lines
102  if (ch_in == '2' || ch_in == '3')
103  {
104  *p_serial << PMS ("Cannot read A/D channel ") << ch_in << endl;
105  }
106  // The characters from 0 through 9 or A through F mean read A/D channels;
107  // characters 'X', 'Y', and 'Z' mean read accelerations; 'L' and 'M' are
108  // commands to auto-balance Wheatstone bridges
109  else if ((ch_in >= '0' && ch_in <= '9') || (ch_in >= 'A' && ch_in <= 'F')
110  || (ch_in >= 'X' && ch_in <= 'Z') || (ch_in >= 'x' && ch_in <= 'z'))
111  {
112  if (oversampling > 1)
113  {
114  *p_serial << p_poly->get_data_oversampled (ch_in, oversampling)
115  << endl;
116  }
117  else
118  {
119  *p_serial << p_poly->get_data (ch_in) << endl;
120  }
121  }
122 
123  // An 'L' or an 'M' mean that we should auto-balance a strain bridge
124  else if (ch_in == 'L' || ch_in == 'M')
125  {
126  p_poly->strain_auto_balance (ch_in - 'K'); // Channel is 1 or 2
127  }
128 
129  // The '#' and '$' are used to set D/A output 1 or 2 respectively
130  else if (ch_in == '#' || ch_in == '$')
131  {
132  *p_serial << "Set D/A " << (uint8_t)(ch_in - '"') << " to: ";
133  *p_serial >> temp_num;
134  p_serial->getchar ();
135  *p_serial << temp_num << endl;
136  p_poly->set_DAC ((uint8_t)(ch_in - '"'), temp_num);
137  }
138 
139  // An uppercase 'O' asks how many times to oversample; 0 turns it off
140  else if (ch_in == 'O')
141  {
142  *p_serial >> oversampling;
143  p_serial->getchar ();
144  *p_serial << "Oversampling ";
145  if (oversampling)
146  {
147  *p_serial << oversampling << " times";
148  }
149  else
150  {
151  *p_serial << " off";
152  }
153  *p_serial << endl;
154  }
155 
156  // The 'd' command means dump the task stacks onto the user interface
157  else if (ch_in == 'd')
158  {
159  print_task_stacks (p_serial);
160  }
161 
162  // Typing 's' requests a task and share status table
163  else if (ch_in == 's')
164  {
165  *p_serial << "Status at " << get_tick_time () << " sec:" << endl;
166  print_task_list (p_serial);
167  *p_serial << endl;
168  p_daq_UI_command_queue->print_all_shares (p_serial);
169  }
170 
171  // The 'a' character runs an I2C accelerometer test
172  else if (ch_in == 'a')
173  {
174  *p_serial << "Accel: " << p_poly->get_data ('X') << ','
175  << p_poly->get_data ('Y') << ','
176  << p_poly->get_data ('Z') << endl;
177  }
178 
179  // Typing 'r' tests Roman numeral printing because it's SO important
180  else if (ch_in == 'r')
181  {
182  *p_serial << "Acceleratio est: " << fortran
183  << p_poly->get_data ('X') << ','
184  << p_poly->get_data ('Y') << ','
185  << p_poly->get_data ('Z') << dec << endl;
186  }
187 
188  // Typing uppercase 'I' means get an IR temperature
189  else if (ch_in == 'I')
190  {
191  *p_serial << "IR: " << p_poly->get_IR_temperature () << endl;
192  }
193 
194  // Typing '@' means scan the I2C bus for sensors
195  else if (ch_in == '@')
196  {
197  *p_serial << "I2C Bus:" << endl;
198  p_poly->scan_I2C_bus (p_serial);
199  }
200 
201  // Typing 'T' gets the time in seconds since the RTOS started up
202  else if (ch_in == 'T')
203  {
204  *p_serial << get_tick_time () << endl;
205  }
206 
207  // The 'h' or question mark characters request a help display
208  else if (ch_in == 'h' || ch_in == '?')
209  {
210  show_help ();
211  }
212 
213  // Return a version string if requested
214  else if (ch_in == 'v')
215  {
216  *p_serial << VERSION_STRING << endl;
217  }
218 
219  // // Character number 3 is a control-C which asks for processor reset
220  // else if (ch_in == 3)
221  // {
222  // *p_serial << "Resetting CPU...Aaarrgh, can't do it" << endl;
223  // chThdSleepMilliseconds (100);
224  // NVIC_SystemReset ();
225  // }
226 
227  // If any other character is entered, ask "What's That Function?"
228  else
229  {
230  if (ch_in < ' ')
231  {
232  *p_serial << '(' << (uint8_t)ch_in << PMS ("): WTF?");
233  }
234  else
235  {
236  *p_serial << ch_in << PMS (": WTF?");
237  }
238  *p_serial << endl;
239  }
240  } // If character entered by user
241 
242  // Print any characters found in the main print queue
243  if (p_main_text_queue->check_for_char ())
244  {
245  p_serial->putchar (p_main_text_queue->getchar ());
246  }
247 
248  delay_ms (1);
249 
250  runs++; // Track how many runs through the loop
251  } // for (;;)
252 }
