72 void task_data_acq::run_daq_command (char ch_in)
73 {
74  // If the command is from 0 through 15, take an A/D conversion, unless it's
75  // using a channel which isn't safe to read
76  if (ch_in == '2' || ch_in == '3')
77  {
78  *p_serial << "Not reading A/D ch. " << ch_in
79  << "; it kills a serial port" << endl;
80  }
81  else if (ch_in >= '0' && ch_in <= '9')
82  {
83  last_data_channel = ch_in;
84  *p_serial << my_poly->get_data (ch_in) << endl;
85  }
86  else if (ch_in >= 'A' && ch_in <= 'F')
87  {
88  last_data_channel = ch_in;
89  *p_serial << my_poly->get_data (ch_in) << endl;
90  }
91 
92  // The characters 'X', 'Y', and 'Z' read accelerations
93  else if (ch_in >= 'X' && ch_in <= 'Z')
94  {
95 // my_poly->set_accel_range (MMA_RANGE_8g); /////////////////////////////////////////////
96 
97  *p_serial << my_poly->get_accel (ch_in - 'X') << endl;
98  }
99 
100  // If we get an 'L' or 'M' character, try to balance a bridge
101  else if (ch_in == 'L')
102  {
103  my_poly->strain_auto_balance (1, 2047);
104  }
105  else if (ch_in == 'M')
106  {
107  my_poly->strain_auto_balance (2, 2047);
108  }
109 }
