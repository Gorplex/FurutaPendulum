39 {

97 /** @brief Get and return raw data from one data channel.
98  * @details This method acquires data from one of the available data channels and
99  * returns it. Channels may include an A/D converter, an accelerometer axis,
100  * or whatever other data source may be configured here.
101  * @param command A one-character command such as @c '6' to get A/D channel 6 data
102  * @return The raw data read from the given channel
103  */
104 
105 int16_t polydaq2::get_data (char command)
106 {
107  // If the command is from '0' through 'A', take an A/D conversion
108  if (command >= '0' && command <= '9')
109  {
110  return (read_ADC (command - '0'));
111  }
112  else if (command >= 'A' && command <= 'F')
113  {
114  return (read_ADC (command - ('A' - 10)));
115  }
116 
117  // The characters 'X', 'Y', and 'Z' read accelerations
118  else if (command >= 'X' && command <= 'Z')
119  {
120  return (get_accel (command - 'X'));
121  }
122 
123  // The characters 'x', 'y', and 'z' read channels from the external accelerometer
124  else if (command >= 'x' && command <= 'z')
125  {
126  return (get_ext_accel (command - 'x'));
127  }
128 
129  // Anything else is baloney, so return zero
130  else
131  {
132  return 0;
133  }
134 }

250 /** @brief Tolerance within which strain gauge balancing is considered acceptable.
251  * @details When a strain gauge bridge is balanced by using the D/A, it is often not
252  * possible to get the zero-strain amplified bridge output to be exactly the
253  * desired value. This tolerance defines a range within which the balancing
254  * is considered good enough; the amplified output can be higher or lower
255  * than the desired value by this amount and the balance will be considered
256  * good enough.
257  */
258 const uint16_t strain_balance_tolerance = 10;
259 
260 /** @brief Number of iterations to make when attempting to balance the bridge.
261  * @details This number specifies how many iterations will be made by the bridge
262  * balancer. The automatic balancing process is a sort of successive
263  * approximation beginning at half the full-scale value, but due to noise
264  * things can get wonky near the end of the approximation process, so we
265  * like to give lots of chances to hit the right value.
266  */
267 const uint8_t strain_balance_retries = 32;

271 /** @brief Automatically zero a strain gauge bridge using the D/A converter
272  * @details This method uses the D/A converter in the STM32 to balance ("zero") a
273  * strain gauge bridge. Each strain gauge bridge has a D/A converter output
274  * connected to one side through a large (20K nominal) resistor; the voltage
275  * from the D/A is used to tweak the bridge output so that the zero-strain
276  * reading is as specified in @c desired_set. A typical value for
277  * @c desired_set is 2047, which is half the full scale for a 12-bit A/D
278  * converter; such a zero setting allows positive and negative strains to be
279  * measured. It is assumed that strain is zero when this method is called.
280  * @param channel The strain bridge channel to balance ("zero"), either 1 or 2
281  * @param desired_set The A/D reading which is desired from the bridge amplifier
282  * when strain is zero (default value: ADC_MAX_OUTPUT / 2)
283  * @return The zero-strain reading which was actually achieved; this may be slightly
284  * off the desired value if the system is noisy. If it isn't possible to
285  * balance the bridge, (-1) is returned
286  */
287 
288 int16_t polydaq2::strain_auto_balance (uint8_t channel, uint16_t desired_set)
289 {
290  int16_t adc_reading; // Most recent reading from A/D
291  int16_t current_DAC; // Most recent value sent to the D/A
292  int16_t error; // Difference between desired and actual
293  uint16_t halfhalf = DAC_MAX_OUTPUT / 4; // Amount to try for successive approx.
294  uint8_t try_count = 0; // Keep track of how many tries are made
295 
296  // Only do anything if the channel number is 1 or 2
297  if (channel == 1 || channel == 2)
298  {
299  // Use a sort of successive approximation, beginning at half of full scale
300  current_DAC = DAC_MAX_OUTPUT / 2;
301 
302  if (p_serial) *p_serial << "Balance: Count, D/A, A/D, Error" << endl;
303  do
304  {
305  // Set the D/A to our best guess and wait a moment
306  set_DAC (channel, (uint16_t)(current_DAC));
307  vTaskDelay (1);
308 
309  // Find how far from the desired value is our current output
310  adc_reading = strain_raw_oversampled (channel, 16);
311  error = desired_set - adc_reading;
312 
313  // Print the current status
314  if (p_serial) *p_serial << try_count << ',' << current_DAC
315  << ',' << adc_reading << ',' << error << endl;
316 
317  // If we're within tolerance, stop here
318  if (error < strain_balance_tolerance && error > -strain_balance_tolerance)
319  {
320  if (p_serial) *p_serial << "Strain bridge " << channel
321  << " balanced to " << adc_reading << endl;
322  return (adc_reading);
323  }
324 
325  // The DAC's are wired to opposite sides of the bridges, oops...
326  if (channel == 1)
327  {
328  error = -error;
329  }
330 
331  // Do the successive approximation thing
332  if (error > 0)
333  {
334  current_DAC += halfhalf;
335  }
336  else
337  {
338  current_DAC -= halfhalf;
339  }
340  if (halfhalf > 1)
341  {
342  halfhalf /= 2;
343  }
344 
345  // Saturate the DAC output to the range of values the DAC can handle
346  if (current_DAC < 0)
347  {
348  current_DAC = 0;
349  }
350  else if (current_DAC > DAC_MAX_OUTPUT)
351  {
352  current_DAC = DAC_MAX_OUTPUT;
353  }
354  vTaskDelay (1);
355  }
356  while (try_count++ < strain_balance_retries);
357 
358  if (p_serial) *p_serial << "Error: Can't balance strain bridge "
359  << channel << " to " << desired_set << ", best try "
360  << adc_reading << endl;
361  return (adc_reading);
362  }
363 
364  // Return something lame if we get here (we shouldn't have; channel was invalid)
365  return (-1);
366 }

417 /** @brief Initialize the SD card's LED to be used with PWM for fancy effects.
418  * @details This method initializes the LED which is used to indicate SD card status
419  * (or to indicate something else if desired). The LED is connected to a
420  * timer and GPIO pin specified in constants that are given in @c polydaq2.h.
421  */
422 
423 void polydaq2::init_SD_card_LED (void)
424 {
425  // Enable the clock to the GPIO port
426  RCC_AHB1PeriphClockCmd (SD_LED_CLOCK, ENABLE);
427 
428  // Initialize the I/O port pin being used
429  GPIO_InitTypeDef GPIO_InitStruct;
430  GPIO_InitStruct.GPIO_Pin = SD_LED_PIN;
431  GPIO_InitStruct.GPIO_Mode = GPIO_Mode_AF;
432  GPIO_InitStruct.GPIO_Speed = GPIO_Speed_50MHz;
433  GPIO_InitStruct.GPIO_OType = GPIO_OType_PP;
434  GPIO_InitStruct.GPIO_PuPd = GPIO_PuPd_UP;
435  GPIO_Init (SD_LED_PORT, &GPIO_InitStruct);
436 
437  // Connect the port pin to its alternate source, which is the PWM
438  GPIO_PinAFConfig (SD_LED_PORT, SD_LED_SOURCE, GPIO_AF_TIM4);
439 
440  // Enable the clock to the timer which will be used by the PWM
441  RCC_APB1PeriphClockCmd (SD_LED_TIMER_CLOCK, ENABLE);
442 
443  // Compute the timer prescaler value. FIXME: Get that magic 21M out of here
444  uint32_t PrescalerValue = (uint16_t)((SystemCoreClock / 2) / 21000000) - 1;
445 
446  // Set up the time base for the timer
447  TIM_TimeBaseInitTypeDef TimeBaseStruct;
448  TimeBaseStruct.TIM_Period = SD_CARD_LED_MAX_PWM;
449  TimeBaseStruct.TIM_Prescaler = PrescalerValue;
450  TimeBaseStruct.TIM_ClockDivision = 0;
451  TimeBaseStruct.TIM_CounterMode = TIM_CounterMode_Up;
452  TIM_TimeBaseInit (SD_LED_TIMER, &TimeBaseStruct);
453 
454  // Configure the output compare used by the PWM
455  TIM_OCInitTypeDef OCInitStruct;
456  OCInitStruct.TIM_OCMode = TIM_OCMode_PWM1;
457  OCInitStruct.TIM_OutputState = TIM_OutputState_Enable;
458  OCInitStruct.TIM_Pulse = 0;
459  OCInitStruct.TIM_OCPolarity = TIM_OCPolarity_High;
460 
461  // PWM1 Mode configuration for Channel X (GPIOY Pin Z)
462  SD_LED_OC_INIT_FN (SD_LED_TIMER, &OCInitStruct);
463  SD_LED_OC_PRNF (SD_LED_TIMER, TIM_OCPreload_Enable);
464 
465  // Set the initial duty cycle to zero
466  SD_LED_SET_DUTY (SD_LED_TIMER, 0);
467 
468  // Enable the timer; this really gets things going
469  TIM_Cmd (SD_LED_TIMER, ENABLE);
470 }
