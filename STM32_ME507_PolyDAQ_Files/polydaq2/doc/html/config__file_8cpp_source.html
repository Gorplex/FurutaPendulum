93 /** @brief Skip whitespace characters and comments until data is found.
94  * @details This method reads characters, skipping whitespace (spaces and returns)
95  * and ignoring lines which are comments, until it has found the first
96  * character on the next active (non-comment) line. That first character is
97  * returned, and the file pointer is left pointing to the character after
98  * that first character.
99  * @return The first non-comment character found, or (-1) if error or end of file
100  */
101 
102 int16_t config_file::skip_to_next_line (void)
103 {
104  char a_ch;
105 
106  // Skip the rest of the current line
107  skip_to_EOL ();
108 
109  // Grab characters; respond to each appropriately until we get to the desired first
110  // character on a line or we run out of characters in the file
111  while (true)
112  {
113  a_ch = p_card->peek ();
114 
115  switch (a_ch)
116  {
117  // If a comment character is seen, skip the rest of this line too
118  case (CFG_COMMENT_CHAR):
119  skip_to_EOL ();
120  break;
121 
122  // The carriage return should be right before a linefeed which ends the
123  // line; just go on to the next character
124  case ('\r'):
125 
126  // A carriage return means we need to go around the loop and examine the
127  // next character to be found
128  case (CFG_END_OF_LINE):
129  p_card->getchar ();
130  break;
131 
132  // A (0xFF) means that we've come to the end of the file or an error
133  case (0xFF):
134  return (-1);
135 
136  // Any other character should be the start of usable data
137  default:
138  return ((char)a_ch);
139  };
140  }
141 }

161 /** @brief Read one boolean value from a configuration file.
162  * @details This method reads a Boolean value from the configuration file. The value
163  * may be represented by various symbols; anything beginning with @c [YyTt]
164  * is taken as meaning true; anything beginning with @c [NnFf] is taken as
165  * false; if the first letter is @c [Oo] then the second letter is checked,
166  * as we've probably got @c On or @c Off.
167  * @param a_char The first character of the item, often found by skip_to_next_line()
168  * @return 0 for false, 1 for true, and 0xFF for undetermined (not a valid Boolean)
169  */
170 
171 uint8_t config_file::read_bool (uint16_t a_char)
172 {
173  uint8_t to_return = 0xFF; // Value to return, error by default
174 
175  switch (a_char)
176  {
177  // If we have the first letter in an obvious true word, set the return value
178  case 'Y':
179  case 'y':
180  case 'T':
181  case 't':
182  to_return = 1;
183  break;
184 
185  // If we have the first letter in what must be false, set the return value
186  case 'N':
187  case 'n':
188  case 'F':
189  case 'f':
190  to_return = 0;
191  break;
192 
193  // If we have an 'O' or an 'o', it could be on or off; find out which one
194  // by looking at the second character
195  case 'O':
196  case 'o':
197  a_char = p_card->getchar ();
198 
199  if (a_char == 'N' || a_char == 'n')
200  {
201  to_return = 1;
202  }
203  else if (a_char == 'F' || a_char == 'f')
204  {
205  to_return = 0;
206  }
207  else
208  {
209  to_return = 0xFF;
210  }
211  break;
212 
213  // Any other character in this location means we're reading junk; return an
214  // error code
215  default:
216  to_return = 0xFF;
217  break;
218  };
219 
220  // Return the value
221  return (to_return);
222 }
