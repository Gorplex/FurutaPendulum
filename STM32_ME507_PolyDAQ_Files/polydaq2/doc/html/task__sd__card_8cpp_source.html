82 void task_sd_card::run (void)
83 {
84 // TickType_t ticks_last_save = 0; // Saves last time data saved to SD card
85 // TickType_t ticks_now; // Temporary storage for current time
86 
87  // Create an SD card object
88  sd_card* p_sdcard = new sd_card (p_serial);
89 
90  // We also need a configuration file reading object
91  logger_config* p_logconf = new logger_config (p_sdcard, p_serial);
92 
93  // The logging file will need a number to make automatic names work
94  uint16_t fnum;
95 
96  // Run the finite state machine in this infinite loop
97  for (;;)
98  {
99  switch (state)
100  {
101  //------------------- State 0: Wait for a card ----------------------------
102  case (0):
103  // Check if a card can be detected
104  if (SD_Detect () == SD_PRESENT)
105  {
106  SD_TSK_DBG (PMS ("SD card detected") << endl);
107  transition_to (1);
108  }
109  else
110  {
111  // Put the LED in heartbeat mode
112  p_led_command_queue->put ('H');
113 
114  // Wait a second before looking again
115  delay_ms (1000);
116  }
117  break;
118 
119  //------------------- State 1: Mount and open -----------------------------
120  case (1):
121  // If there's no card detected, go back to state 0
122  if (SD_Detect () != SD_PRESENT)
123  {
124  SD_TSK_DBG ("SD card removed" << endl);
125  transition_to (0);
126  }
127  else // There's a card; mount it, open a file, and go to state 2
128  {
129  SD_TSK_DBG (PMS ("Mounting..."));
130  if (p_sdcard->mount () == FR_OK)
131  {
132  transition_to (2);
133  }
134  else // Oops, problems...wait 5 seconds and try again
135  {
136  SD_TSK_DBG (PMS ("Oh Noes! SD card mount problem") << endl);
137  transition_to (0);
138  delay_ms (5000);
139  }
140  SD_TSK_DBG (endl);
141  }
142  break;
143 
144  //------------------- State 2: Read configuration file --------------------
145  case (2):
146  // There's a card; read the configuration file and go to state 3
147  if (SD_Detect () == SD_PRESENT)
148  {
149  // Turn the LED on -- the card is busy
150  p_led_command_queue->put ('M');
151  sd_card_LED = true;
152 
153  // Try to open the configuration file
154  SD_TSK_DBG (PMS ("Reading configuration..."));
155  p_logconf->read ("polydaq2.cfg");
156  SD_TSK_DBG (PMS ("done") << endl);
157 
158  // Done reading the SD card for a little while
159  sd_card_LED = false;
160 
161  // If the configuration file's data is valid, use it
162  if (p_logconf->is_valid ())
163  {
164  p_logger_config->put (p_logconf);
165  p_ticks_per_sd_data->put (p_logconf->get_ticks_per_sample ());
166  SD_TSK_DBG (*p_logconf << endl);
167  transition_to (3);
168  }
169  else // If the configuration is bad, don't save any data
170  {
171  SD_TSK_DBG (PMS ("SD card removed") << endl);
172  p_sdcard->unmount ();
173  p_logger_config->put (NULL);
174  p_led_command_queue->put ('H');
175  transition_to (9);
176  }
177  }
178  else // If there's no card detected, go back to state 0
179  {
180  SD_TSK_DBG ("No can has SD card." << endl);
181  p_logger_config->put (NULL);
182  p_led_command_queue->put ('H');
183  transition_to (0);
184  }
185  break;
186 
187  //------------------- State 3: Open data logging file ---------------------
188  case (3):
189  // If there's no card detected, go back to state 0
190  if (SD_Detect () != SD_PRESENT)
191  {
192  SD_TSK_DBG ("No can has SD card." << endl);
193  p_logger_config->put (NULL);
194  p_led_command_queue->put ('H');
195  transition_to (0);
196  }
197  else // There's a card; open the logging file
198  {
199  // Try to open the data logging file
200  fnum = p_sdcard->open_new_data_file (DATA_F_NAME, DATA_F_EXT);
201  if (fnum == 0xFFFF)
202  {
203  SD_TSK_DBG (PMS ("Oh Noes! Can't open data file") << endl);
204  p_sdcard->unmount ();
205  transition_to (0);
206  delay_ms (5000);
207  }
208  else
209  {
210  *p_sdcard << PMS ("PolyDAQ 2 data file, samples at ")
211  << (p_logger_config->get ())->get_ms_per_sample ()
212  << PMS (" ms") << endl;
213 // ticks_last_save = get_tick_count ();
214 
215  write_header (p_sdcard);
216  transition_to (4);
217  }
218  }
219  break;
220 
221  //------------------- State 4: Wait for and save data ---------------------
222  case (4):
223  // Watch for data which needs to be written
224  while (p_sd_card_text_queue->check_for_char ())
225  {
226  p_sdcard->putchar (p_sd_card_text_queue->getchar ());
227  }
228 
229 // // Check if it's time to flush the text buffer to the SD card
230 // ticks_now = get_tick_count ();
231 // if ((ticks_now - ticks_last_save) > SD_TICKS_PER_SYNC)
232 // {
233 // ticks_last_save = ticks_now;
234 // *p_sdcard << send_now;
235 // }
236 
237  // Check for the card, then let other tasks run
238  if (SD_Detect () != SD_PRESENT)
239  {
240  SD_TSK_DBG (PMS ("SD card removed") << endl);
241  p_sdcard->unmount ();
242  p_logger_config->put (NULL);
243  p_led_command_queue->put ('H');
244  transition_to (0);
245  }
246  delay_ms (1); ///////////////////////////////////////////////////////////
247 // yield ();
248  break;
249 
250  //------------------- State 9: Bad configuration file ---------------------
251  case (9):
252  // Turn the LED off and wait until the card has been removed
253  // Check for the card, then let other tasks run
254  if (SD_Detect () != SD_PRESENT)
255  {
256  transition_to (0);
257  }
258  break;
259 
260  //------------------- Other States: Should never happen -------------------
261  default:
262  transition_to (0);
263  break;
264  }; // switch
265 
266 // ch_in = p_serial->getchar (); // This should block for empty buffer
267 
268  runs++; // Track how many runs through the loop
269  } // for (;;)
270 }

274 /** @brief Write a header showing data items to be logged to a serial device.
275  * @details This method writes a header line which shows the name of each of the
276  * items to be logged. Usually the logging is done to an SD card, and this
277  * line is a header for the logged data. The logger configuration object,
278  * which is pointed to by the shared variable @c p_logger_config, is used
279  * to get the data headers.
280  * @param p_ser_dev A pointer to the serial device to which to write the header
281  */
282 
283 void task_sd_card::write_header (emstream* p_ser_dev)
284 {
285  logger_config* p_the_config; // Data logger configuration object
286  logger_col_cfg* p_col_cfg; // Stores each data column's configuration
287 
288  // If the logger configuration isn't NULL, there's an SD card present
289  if ((p_the_config = p_logger_config->get ()) != NULL)
290  {
291  // Write the current RTOS time down to the millisecond
292  *p_ser_dev << PMS ("Time");
293 
294  // For each configured data item, get the header and write it
295  p_col_cfg = p_the_config->get_first_channel ();
296  while (p_col_cfg != NULL)
297  {
298  *p_ser_dev << ',' << '"' << p_col_cfg->p_label << '"';
299  p_col_cfg = p_the_config->get_next_channel ();
300  }
301  *p_ser_dev << '\n';
302  }
303 }
