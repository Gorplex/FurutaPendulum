64 void task_sd_daq::acquire_sd_data (logger_config* p_log_config)
65 {
66  logger_col_cfg* p_col_cfg; // Stores each data column's configuration
67  int16_t data; // Data which is read from the PolyDAQ
68 
69  // Turn the LED on -- the card is busy
70  p_led_command_queue->put ('Y');
71 
72  // Write the current RTOS time down to the millisecond
73  *p_sd_card_text_queue << get_tick_time ();
74 
75  // For each configured data item, get the data and write it
76  p_col_cfg = p_log_config->get_first_channel ();
77  while (p_col_cfg != NULL)
78  {
79  data = my_poly->get_data (p_col_cfg->command);
80  *p_sd_card_text_queue << ','
81  << (float)data * p_col_cfg->slope + p_col_cfg->offset;
82 // << '"' << p_col_cfg->command << '"';
83  p_col_cfg = p_log_config->get_next_channel ();
84  }
85  *p_sd_card_text_queue << DAQ_EOL;
86 
87  // Turn the LED off -- the card is done for now
88  p_led_command_queue->put ('N');
89 }

98 void task_sd_daq::run (void)
99 {
100  logger_config* p_the_config; // Data logger configuration object
101 
102  // This counter is used to run through the for (;;) loop at precise intervals
103  TickType_t LastWakeTime = xTaskGetTickCount ();
104 
105  ////////////////////// TESTING FOR MLX90614 PIR THERMOMETER ///////////////////////
106  // Drop the SCL line for a couple of milliseconds to put thermo in SMBus (I2C) mode
107  delay_ms (2);
108  ///////////////////////////////////////////////////////////////////////////////////
109 
110  // Do initializations which must be done with the RTOS already up and running
111  my_poly->initialize ();
112 
113  // In the main loop, wait for a command through the command queue; if a command
114  // is received, do what the commander asked
115  for (;;)
116  {
117  // If the logger configuration isn't NULL, there's an SD card present
118  if ((p_the_config = p_logger_config->get ()) != NULL)
119  {
120  acquire_sd_data (p_the_config);
121  delay_from_for (LastWakeTime, p_the_config->get_ticks_per_sample ());
122  }
123  else
124  {
125  delay_from_for_ms (LastWakeTime, 100); // No card? Wait a bit longer
126  }
127 
128  runs++; // Count runs through the loop
129  }
130 }
